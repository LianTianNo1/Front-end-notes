    # leetcod:206.反转链表

    leetcod:206.反转链表 (opens new window)

    反转两个节点:将n+1的next指向n

    反转多个节点:双指针遍历链表

    重复上述操作。

    1. 双指针一前一后遍历链表
    2. 反转双指针
    /**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var reverseList = function(head) {
    // 双指针遍历链表
    let p1 = head;
    let p2 = null;
    while(p1){
        // 把p1.next先保存下来
        const temp = p1.next;
        // 改变p1的next指向，但是后面又会使用之前保存下来的变量
        // 其实这里就开始在反转链表了,不断的改变next 的指向
        // 等下使用 p2 来接收
        p1.next = p2;
        // p2 也跟着向下面移动，比p1慢了一步
        p2 = p1;
        // p1 指针继续向下移动
        p1 = temp;        
    }
    // 实际返回的是 p2
    // console.log(p2)
    return p2
};

    时间复杂度 O(n)

    空间复杂度O(1)

    这里使用的是一个单个变量 temp

    • 时间: O(n). 逐个推进故 O(n).
    • 空间: O(1). 只用到了常数级额外空间故 O(1).

        const val2 = p2 ? p2.val: 0;
        // 6.进行相加操作

        // 8.个位数添加到新链表上面

        if(p1) p1 = p1.next;
        if(p2) p2 = p2.next;
        p3 = p3.next;
    }
    // 10.判断是否进位
    if(carry){
        p3.next = new ListNode(carry);
    }
    // 11.返回值因为第一次初始化的时候第一个val是0，所以就返回 l3.next

var minWindow = function (s, t) {
  // 左指针
  let left = 0
  // 右指针
  let right = 0
  // 返回结果
  let res = ''
  // 需要包含的字符
  let need = new Map()
  // 需要的字符串放进字典里面
  for (let str of t) {
    // 放进字典之前判断是否字典里面是否存在了，存在了需求就+1
    need.set(str, need.has(str) ? need.get(str) + 1 : 1)
  }
  // 需要的字符个数
  let needType = need.size
  // 移动右指针
  while (right < s.length) {
    // 右指针对应的值
    const rValue = s[right]
    if (need.has(rValue)) {
      // 找到了需求就-1
      need.set(rValue, need.get(rValue) - 1)
      // 当这个字符需求为0的时候，needType 就是总共需要的字符也为就减1
      if (need.get(rValue) === 0) needType--
    }
    // 当移动右指针已经包含了所有的需要的字符了，现在移动左指针缩小范围
    while (needType === 0) {
      // 左指针对应的值
      const newRes = s.substring(left, right + 1)
      if (!res || newRes.length < res.length) res = newRes
      const lValue = s[left]
      // 如果左指针对应的值包含在需求里面，等下要移动左指针，我们就把需求+1
      if (need.has(lValue)) {
        need.set(lValue, need.get(lValue) + 1)
        if (need.get(lValue) === 1) needType++
      }
      // 移动左指针
      left++
    }
    // 移动右指针
    right++
  }
  // 返回结果
  return res
}

 *    this.val = val === undefined ? 0 : val;
 *    this.neighbors = neighbors === undefined ? [] : neighbors;
 * };

/**
 * @param {Node} node
 * @return {Node}

var cloneGraph = function(node) {
    if(!node)return;
    // 这个字典用来判断是否已经遍历过了，还有存放克隆的图
    const visited = new Map();
    // 这里使用优先广度遍历
    // 所以这里使用队列来模拟
    const queue = [node];
    // 第一次把node 添加到Map 中
    visited.set(node,new Node(node.val));
    // 循环判断队列中是否存在节点
    while(queue.length){
        // 先进先出
        const n = queue.shift();
        // 循环邻接点
        (n.neighbors || []).forEach(nextNode=>{  
            // 对没有进行遍历的进行操作
            if(!visited.has(nextNode)){
                // 添加邻接节点
                 queue.push(nextNode);
                //  这里克隆邻接点
                 visited.set(nextNode,new Node(nextNode.val));
            };
            // 添加边的关系
            visited.get(n).neighbors.push(visited.get(nextNode))
        });
    }
    return visited.get(node)
};

Array.prototype.insertSort = function () {
  // 从第二个数开始往前比。
  for (let i = 1; i < this.length; i++) {
    // 先保存下第二个数
    let temp = this[i]
    // j往前走，因为当前的数要和前面的所有数比较
    let j = i
    while (j > 0) {
      // 如果前面数大于保存的temp
      if (this[j - 1] > temp) {
        // 前面的向后走
        this[j] = this[j - 1]
      } else {
        // 如果是小于等于跳出循环
        break
      }
      // 继续向先找大的，遇到小的跳出循环，这个j就是temp要插入的地方
      j--
    }
    // 把temp插入到找到的j的位置，j的位置就是前面后排的元素现在空着
    this[j] = temp
  }

    # 4.归并排序

    归并排序的思路

    分:把数组劈成两半，再递归地对子数组进行"分”操作，直到分成一个个单独的数。

    合:把两个数合并为有序数组，再对有序数组进行合并，直到全部子数组合并为一-个完整数组。

    合并两个有序数组

    新建一个空数组res,用于存放最终排序后的数组。

    比较两个有序数组的头部，较小者出队并推入res中。

    如果两个数组还有值,就重复第二步。

    Array.prototype.mergeSort = function () {
  // 分 合
  const rec = (arr) => {
    // 一直二分，如果是一就返回arr
    if (arr.length === 1) return arr
    // 找到中间的值，奇数值向下取整
    const mid = Math.floor(arr.length / 2)
    // 二分
    const leftArray = arr.slice(0, mid)
    const rightArray = arr.slice(mid, arr.length)
    // 不断的二分，直到每个一个数都是子数组，然后进行对比头数组入对，并返回了回来
    const orderLeft = rec(leftArray)
    const orderRight = rec(rightArray)
    // 用来合的res数组，把返回orderLeft，orderRight循环出队添加进res
    const res = []
    while (orderLeft.length || orderRight.length) {
      // 左右都有的话，就比较两个的头数组，小的就行放进合并的数组
      if (orderLeft.length && orderRight.length) {
        res.push(
          orderLeft[0] < orderRight[0] ? orderLeft.shift() : orderRight.shift()
        )
        // 如果两个数组还有直接出队放进合并的数组
      } else if (orderLeft.length) {
        res.push(orderLeft.shift())
      } else if (orderRight.length) {
        res.push(orderRight.shift())
      }
    }
    // 返回合并的数组，给上个递归用，要么就是最终的结果
    return res
  }
  // 获取到合并的数组
  const res = rec(this)
  // 重新赋值给arr
  res.forEach((value, index) => {
    this[index] = value
  })
}

    快速排序的思路

    分区:从数组中任意选择一一个“基准”，所有比基准小的元素放在基准前面，比基准大的元素放在基准的后面。

    递归:递归地对基准前后的子数组进行分区。

    Array.prototype.quickSort = function () {

  const rec = (arr) => {
    if (arr.length === 0 || arr.length === 1) return arr
    // 分左边和右边
    const left = []
    const right = []
    // 以第一个为基准
    const mid = arr[0]
    // 从第二个开始
    for (let i = 1; i < arr.length; i++) {
      // 比基准小的加入到right，比基准大的加入到left
      if (arr[i] < mid) {
        left.push(arr[i])
      } else {
        right.push(arr[i])
      }
    }
    // 组合成 left mid right 为上一次的递归使用，或者最终返回
    return [...rec(left), mid, ...rec(right)]
  }
  const res = rec(this)
  // 赋值操作
  res.forEach((value, index) => (this[index] = value))
}

    从数组的中间元素开始，如果中间元素正好是目标值，则搜索结束。

    如果目标值大于或者小于中间元素，则在大于或小于中间元素的那一半数组中搜索。

    Array.prototype.binarySearch = function (item) {

      // 如果比中间值小，说明在左边，改变最大范围
    } else if (item < midELement) {
      // 这种情况就说明巧合要搜索的值就是中间值
      high = mid - 1
    } else {
      return mid
    }
  }
  return -1
}

    # 21. 合并两个有序链表 (opens new window)

    解题思路

    与归并排序中的合并两个有序数组很相似。

    将数组替换成链表就能解此题。

    解题步骤

    新建一个新链表，作为返回结果。

    用指针遍历两个有序链表，并比较两个链表的当前节点，

    较小者先接入新链表，并将指针后移一步。

    链表遍历结束，返回新链表。

    /**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} list1
 * @param {ListNode} list2
 * @return {ListNode}
 */
var mergeTwoLists = function(list1, list2) {
  const res = new ListNode(0);
  let p = res;
  let p1 = list1;
  let p2 = list2;
  while(p1 && p2){
    if(p1.val < p2.val){
      p.next = p1;
      p1 = p1.next;
    }else{
      p.next = p2;
      p2 = p2.next;
    }
    p = p.next
  }
  if(p1) p.next = p1;
  if(p2) p.next = p2;
  return res.next
};

    解题思路

    这不就是二分搜索嘛!

    从数组的中间元素开始，如果中间元素正好是目标值，则搜索过程结束。

    /** 
 * Forward declaration of guess API.
 * @param {number} num   your guess
 * @return 	            -1 if num is lower than the guess number
 *			             1 if num is higher than the guess number
 *                       otherwise return 0
 * var guess = function(num) {}
 */

/**
 * @param {number} n
 * @return {number}
 */
var guessNumber = function(n) {
    let low = 1, high = n;
    while(low<=high){
      const mid = Math.floor((low+high)/2);
      const res = guess(mid);
      if(res===-1) high = mid - 1;
      else if(res===1) low = mid + 1;
      else return mid
    }
};

    # 分而治之

    分而治之是什么?

    分而治之是算法设计中的一种方法。

    它将一个问题==分==成==多个和原问题相似的小问题==，递归解决小问题,

    再将结果==合==并以解决原来的问题。

    # 场景

    场景一:归并排序

    分:把数组从中间一分为二。

    解:递归地对两个子数组进行归并排序。

    合:合并有序子数组。

    场景二:快速排序

    分:选基准，按基准把数组分成两个子数组。

    解:递归地对两个子数组进行快速排序。

    合:对两个子数组进行合并。

    # 374. 猜数字大小(使用分而治之思想) (opens new window)

    解题思路

    • 二分搜索，同样具备“分、解、合”的特性。
    • 考虑选择分而治之。

    解题步骤

    分:计算中间元素，分割数组。

    解:递归地在较大或者较小子数组进行二分搜索。

    合:不需要此步，因为在子数组中搜到就返回了。

    /** 
 * Forward declaration of guess API.
 * @param {number} num   your guess
 * @return 	            -1 if num is lower than the guess number
 *			             1 if num is higher than the guess number
 *                       otherwise return 0
 * var guess = function(num) {}
 */

/**
 * @param {number} n
 * @return {number}
 */
var guessNumber = function(n) {
    const rec = (low,high)=>{
      if(low>high)return
      const mid = Math.floor((low+high)/2)
      const res = guess(mid)
      if(res === 0) {
        return mid
      }else if(res === -1){
        return rec(low,mid-1)
      }else if(res === 1){
        return rec(mid+1,high)
      }
    }
    return rec(1,n)
};

    # 226. 翻转二叉树 (opens new window)

    解题思路

    先翻转左右子树，再将子树换个位置。

    符合"分、解、合”特性。

    解题步骤

    分:获取左右子树。

    解:递归地翻转左右子树。

 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {TreeNode}
 */
var invertTree = function(root) {
  // 为空直接返回null
  if(!root)return null
  // 返回给上一个递归left，right,或者最终结果,left和right进行交换
  return{
    val : root.val,
    left : invertTree(root.right),
    right: invertTree(root.left)
  }

    # 100. 相同的树 (opens new window)

    解题思路

    两个树:根节点的值相同，左子树相同,右子树相同。

    符合“分、解、合”特性。

    考虑选择分而治之。

    解题步骤

    分:获取两个树的左子树和右子树。

    解:递归地判断两个树的左子树是否相同，右子树是否相同。

    合:将上述结果合并，如果根节点的值也相同，树就相同。

    /**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} p
 * @param {TreeNode} q
 * @return {boolean}
 */
var isSameTree = function(p, q) {
  // 两个都为null返回true，直接返回，或作为左子树和右子树的结果返回
  if(!p && !q) return true;
  // 在两个都存在，并且两个值都相等，还有所有左子树和右子树的递归结果
  if(p && q && p.val === q.val && isSameTree(p.left, q.left) && isSameTree(p.right, q.right)){
    return true;
  }
  return false;
};

    时间复杂度:O(N)

    空间复杂度:O(N) 或者等于树的高度O(logN)

    # 101. 对称二叉树 (opens new window)

    解题思路

    转化为:左右子树是否镜像。

    分解为:树1的左子树和树2的右子树是否镜像,

    树1的右子树和树2的左子树是否镜像。

    符合"分、解、合”特性，考虑选择分而治之。

    解题步骤

    分:获取两个树的左子树和右子树。

    解:递归地判断树1的左子树和树2的右子树是否镜像，树1的右子树和树2的左子树是否镜像。

    合:如果上述都成立，且根节点值也相同,两个树就镜像。

    /**
 * Definition for a binary tree node.
 * function TreeNode(val, left, right) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.left = (left===undefined ? null : left)
 *     this.right = (right===undefined ? null : right)
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isSymmetric = function(root) {
  // 空节点直接返回true
  if(!root)return true
  // 用来递归判断是否是镜像
  const isMirror = (l,r) => {
    // 如果两个空返回true给上个递归
    if(!l && !r) return true;
    // 判断传入的两个左右根节点是否相等，然后判断左根节点的左子树和右根节点的右子树是否相等，左根节点的右子树和右根节点的左子树是否相等（这两个都是递归的判断子树是否相等）
    if(l && r && l.val === r.val &&
    isMirror(l.left,r.right) &&
    isMirror(l.right, r.left)
    ){
      return true;
    }
    return false;
  }
  return isMirror(root.left, root.right)
};

    时间复杂度:O(N)

    空间复杂度:O(N) 最好树的高度O(h) 均匀分布的话O(logN)

    # 70. 爬楼梯 (opens new window)

    解题思路

    爬到第n阶可以在第n-1阶爬1个台阶，或者在第n-2阶爬2个台阶。

    F(n)= F(n-1) + F(n-2)。

    使用动态规划。

    解题步骤

    定义子问题: F(n) = F(n-1) + F(n-2)。

    反复执行:从2循环到n,执行上述公式。(如果是1就只有一种方法了，n-2就等负数...)

    /**
 * @param {number} n
 * @return {number}
 */
var climbStairs = function(n) {
  if(n<2)return 1
  const dp = [1,1]
  for(let i = 2; i <= n; i++){
    dp[i] = dp[i-2] + dp[i-1]
  }
  return dp[n]
};

    /**
 * @param {number} n
 * @return {number}
 */
var climbStairs = function(n) {
  if(n===0)return 0
  const dp = []
  // 爬一个有一种方法
  dp[1] = 1
  // 爬两个有两种方法
  dp[2] = 2
  // 从第三个开始循环
  // 满足 dp[i] = dp[i-2] + dp[i-1]
  // 例如 3 = dp[1] + dp[2] = 1 + 2
  for(let i = 3; i <= n; i++){
    dp[i] = dp[i-2] + dp[i-1]
  }
  return dp[n]
};
